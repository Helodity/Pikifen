states_ignoring_hazard = teleporting_to_leader; teleported_to_leader

init {
    set_var follow_upon_idling false
}

script {
    idling {
        on_enter {
            stop_particles
            start_particles glow_pikmin
            custom_actions_after
            if $follow_upon_idling = true
                set_var follow_upon_idling false
                set_var teleport_timer 9999
                focus_on_id $leader_id
                follow_mob_as_leader focus true
            else
                get_random_float temp 1.0 1.5
                set_var teleport_timer $temp
            end_if
        }
        on_tick {
            get_misc_info temp delta_t
            calculate teleport_timer $teleport_timer - $temp
            if $teleport_timer <= 0

                //Decide which leader to teleport to.
                get_mob_info cur_team self team
                clear_var player_leaders
                clear_var player_leaders_glows

                get_misc_info temp player_1_leader_id
                add_list_item player_leaders $player_leaders $temp
                get_misc_info temp player_2_leader_id
                add_list_item player_leaders $player_leaders $temp
                get_misc_info temp player_3_leader_id
                add_list_item player_leaders $player_leaders $temp
                get_misc_info temp player_4_leader_id
                add_list_item player_leaders $player_leaders $temp

                for_each nr id $player_leaders
                    if $id != 0
                        focus_on_id $id
                        get_mob_info temp focus team
                        if $temp = $cur_team
                            //This is one of our active leaders.
                            get_leader_pikmin_count temp focus glow_pikmin
                            add_list_item player_leaders_glows $player_leaders_glows $temp
                        else
                            //This leader is not on our team.
                            add_list_item player_leaders_glows $player_leaders_glows -1
                        end_if
                    else
                        //This player has no leader.
                        add_list_item player_leaders_glows $player_leaders_glows -1
                    end_if
                end_for_each

                set_var neediest_leader_nr 0
                set_var neediest_leader_amount 99999
                for_each nr glows $player_leaders_glows
                    if $glows != -1
                        if $glows < $neediest_leader_amount
                            //This leader needs Glow Pikmin more than the previous best.
                            set_var neediest_leader_nr $nr
                            set_var neediest_leader_amount $glows
                        end_if
                    end_if
                end_for_each

                //Teleport, if possible.
                if $neediest_leader_nr != 0
                    get_list_item temp $player_leaders $neediest_leader_nr
                    set_var leader_id $temp
                    set_state teleporting_to_leader
                end_if

            end_if
        }
    }
    in_group_chasing {
        on_active_leader_changed {
            get_mob_info leader_id trigger id
            set_state teleporting_to_leader
        }
    }
    in_group_stopped {
        on_active_leader_changed {
            get_mob_info leader_id trigger id
            set_state teleporting_to_leader
        }
    }
    teleporting_to_leader {
        on_enter {
            set_shadow_visibility false
            set_animation teleport_start
            stop_particles
            start_particles glow_pikmin_teleport_out
            play_sound teleport_out
        }
        on_animation_end {
            set_var cur_leader_a 9999
            focus_on_id $leader_id
            get_mob_info cur_leader_a focus angle
            if $cur_leader_a != 9999
                //The new leader is valid.
                get_mob_info cur_leader_x focus x
                get_mob_info cur_leader_y focus y
                get_mob_info cur_leader_z focus z
                get_coordinates_from_angle temp_x temp_y $cur_leader_a -40
                calculate cur_leader_x $cur_leader_x + $temp_x
                calculate cur_leader_y $cur_leader_y + $temp_y
                get_random_float temp_x -30 30
                get_random_float temp_y -30 30
                calculate cur_leader_x $cur_leader_x + $temp_x
                calculate cur_leader_y $cur_leader_y + $temp_y
                teleport_to_absolute $cur_leader_x $cur_leader_y $cur_leader_z
            else
                //The new leader is invalid.
            end_if
            set_state teleported_to_leader
        }
    }
    teleported_to_leader {
        on_enter {
            set_animation teleport_end
            stop_particles
            start_particles glow_pikmin_teleport_in
            play_sound teleport_in
        }
        on_animation_end {
            set_shadow_visibility true
            set_var follow_upon_idling true
            set_state idling
        }
    }
    dying {
        on_enter {
            set_timer 1.9
        }
        on_timer {
            start_particles glow_pikmin_pop
        }
    }
    knocked_down_dying {
        on_enter {
            set_timer 0.5
        }
        on_timer {
            start_particles glow_pikmin_pop
        }
    }
    crushed {
        on_enter {
            set_timer 1.2
        }
        on_timer {
            start_particles glow_pikmin_pop
        }
    }
}
