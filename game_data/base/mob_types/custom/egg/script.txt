first_state = stage_1_idling
death_state = destroyed

init {
    set_var contents random
}

ready {
    get_mob_info temp self health_ratio
    if $temp < 0.33
        set_state stage_3_idling
    else_if $temp < 0.66
        set_state stage_2_idling
    end_if
}

script {
    stage_1_idling {
        on_enter {
            set_animation stage_1_idling
        }
        on_damage {
            get_mob_info temp self health_ratio
            if $temp < 0.33
                set_state stage_3_idling
            else_if $temp < 0.66
                set_state stage_2_idling
            end_if
        }
        on_land {
            stop
        }
    }
    stage_2_idling {
        on_enter {
            set_animation stage_2_idling
        }
        on_damage {
            get_mob_info temp self health_ratio
            if $temp < 0.33
                set_state stage_3_idling
            end_if
        }
        on_land {
            stop
        }
    }
    stage_3_idling {
        on_enter {
            set_animation stage_3_idling
        }
        on_land {
            stop
        }
    }
    destroyed {
        on_enter {
            set_animation destroyed
            start_dying
            start_particles egg_shard_explosion
            play_sound crack
            
            if $contents = random
                get_random_float temp 1 100
                if $temp < 55
                    set_var contents nectar
                else_if $temp < 90
                    set_var contents nectar_2
                else_if $temp < 95
                    set_var contents spicy_drop
                else
                    set_var contents bitter_drop
                end_if
            end_if
            
            if $contents = nectar
                spawn nectar
            else_if $contents = nectar_2
                spawn nectar
                spawn nectar
            else_if $contents = spicy_drop
                spawn spicy_drop
            else_if $contents = bitter_drop
                spawn bitter_drop
            end_if
        }
        on_animation_end {
            finish_dying
            delete
        }
    }
}
