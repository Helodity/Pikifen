first_state = idling

script {
    idling {
        on_enter {
            send_message_to_links 0
        }
        on_receive_message {
            get_event_info msg message
            get_list_item temp $msg 1

            if $temp = power
                get_list_item input_power $msg 2
                get_mob_info id trigger id

                get_list_item_number temp $ids $id
                if $temp = 0
                    add_list_item ids $ids $id
                    add_list_item powers $powers $input_power
                else
                    set_list_item powers $powers $input_power $temp
                end_if

                set_var all_off true
                for_each temp temp2 $powers
                    if $temp2 > 0
                        set_var all_off false
                    end_if
                end_for_each

                set_var msg power
                if $all_off = true
                    add_list_item msg $msg 0
                else
                    add_list_item msg $msg 1
                end_if
                send_message_to_links $msg
            end_if
        }
    }
}
