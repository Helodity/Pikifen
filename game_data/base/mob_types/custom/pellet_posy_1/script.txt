first_state = sprout
death_state = dying
states_ignoring_death = reviving

init {
    set_var red false
    set_var yellow false
    set_var blue false
    set_var purple false
    set_var white false
    set_var rock false
    set_var winged false
    set_var ice false

    set_var bloom_time 240
    set_var revival_time 0
    
    set_var cur_color 1
}

ready {
    if $red = true
        add_list_item list $list red
    end_if
    if $yellow = true
        add_list_item list $list yellow
    end_if
    if $blue = true
        add_list_item list $list blue
    end_if
    if $purple = true
        add_list_item list $list purple
    end_if
    if $white = true
        add_list_item list $list white
    end_if
    if $rock = true
        add_list_item list $list rock
    end_if
    if $winged = true
        add_list_item list $list winged
    end_if
    if $ice = true
        add_list_item list $list ice
    end_if

    get_list_size temp $list
    if $temp = 0
        //No colors have been chosen. Use red.
        add_list_item list $list red
    end_if
}

script {
    sprout {
        on_enter {
            set_animation sprout
            set_huntable false
            calculate temp $bloom_time / 2
            max_number temp $temp 1
            set_timer $temp
        }
        on_timer {
            set_state sapling
        }
    }

    sapling {
        on_enter {
            set_animation sapling
            start_particles sparks_small 15 0 150
            calculate temp $bloom_time / 2
            max_number temp $temp 1
            set_timer $temp
        }
        on_timer {
            set_state blooming
        }
    }

    blooming {
        on_enter {
            set_animation blooming
            start_particles sparks_small 15 0 150
        }
        on_animation_end {
            set_huntable true
            get_list_size temp $list
            set_var cur_color $temp
            set_state picking_next_color
        }
    }

    picking_next_color {
        on_enter {
            get_list_size temp $list
            calculate cur_color $cur_color + 1
            if $cur_color > $temp
                set_var cur_color 1
            end_if
            
            set_timer 3
            set_state idling
        }
    }

    idling {
        on_enter {
            get_list_item temp $list $cur_color
            add_to_string anim_name idling_ $temp
            set_animation $anim_name no_restart
        }
        on_damage {
            play_sound squeak
        }
        on_timer {
            set_state picking_next_color
        }
    }

    dying {
        on_enter {
            set_animation dying
            set_huntable false
            start_particles enemy_defeat 0 0 150
            start_particles pellet_posy_1_petals 40 0 150
            start_dying
            get_list_item temp $list $cur_color
            spawn $temp
        }
        on_animation_end {
            set_state reviving
        }
    }

    reviving {
        on_enter {
            set_timer $revival_time
        }
        on_timer {
            get_mob_info temp self max_health
            set_health $temp
            set_state sprout
        }
    }
}
