first_state = sprout
death_state = dying
states_ignoring_death = reviving

init {
    set_var red false
    set_var yellow false
    set_var blue false
    set_var purple false
    set_var white false
    set_var rock false
    set_var winged false
    set_var ice false

    set_var cur_color none
    set_var bloom_time 240
    set_var revival_time 0
}

ready {
    if $red = true
        set_state sprout
    else_if $yellow = true
        set_state sprout
    else_if $blue = true
        set_state sprout
    else_if $purple = true
        set_state sprout
    else_if $white = true
        set_state sprout
    else_if $rock = true
        set_state sprout
    else_if $winged = true
        set_state sprout
    else_if $ice = true
        set_state sprout
    end_if
    
    //If it got here, that means it has no chosen content. Pick red.
    set_var red true
    set_state sprout
}

script {
    sprout {
        on_enter {
            set_animation sprout
            set_huntable false
            calculate temp $bloom_time / 2
            if $temp <= 1
                set_var temp 1
            end_if
            set_timer $temp
        }
        on_timer {
            set_state sapling
        }
    }

    sapling {
        on_enter {
            set_animation sapling
            start_particles sparks_small 15 0 150
            calculate temp $bloom_time / 2
            if $temp <= 1
                set_var temp 1
            end_if
            set_timer $temp
        }
        on_timer {
            set_state blooming
        }
    }

    blooming {
        on_enter {
            set_animation blooming
            start_particles sparks_small 15 0 150
        }
        on_animation_end {
            set_huntable true
            set_var cur_color ice
            set_state picking_next_color
        }
    }

    picking_next_color {
        on_enter {
            set_timer 3
            
            label try_next
            
            if $cur_color = red
                set_var cur_color yellow
            else_if $cur_color = yellow
                set_var cur_color blue
            else_if $cur_color = blue
                set_var cur_color purple
            else_if $cur_color = purple
                set_var cur_color white
            else_if $cur_color = white
                set_var cur_color rock
            else_if $cur_color = rock
                set_var cur_color winged
            else_if $cur_color = winged
                set_var cur_color ice
            else_if $cur_color = ice
                set_var cur_color red
            end_if
            
            if $cur_color = red
                if $red != true
                    goto try_next
                end_if
            else_if $cur_color = yellow
                if $yellow != true
                    goto try_next
                end_if
            else_if $cur_color = blue
                if $blue != true
                    goto try_next
                end_if
            else_if $cur_color = purple
                if $purple != true
                    goto try_next
                end_if
            else_if $cur_color = white
                if $white != true
                    goto try_next
                end_if
            else_if $cur_color = rock
                if $rock != true
                    goto try_next
                end_if
            else_if $cur_color = winged
                if $winged != true
                    goto try_next
                end_if
            else_if $cur_color = ice
                if $ice != true
                    goto try_next
                end_if
            end_if
            
            set_state idling
        }
    }

    idling {
        on_enter {
            if $cur_color = red
                set_animation idling_red no_restart
            else_if $cur_color = yellow
                set_animation idling_yellow no_restart
            else_if $cur_color = blue
                set_animation idling_blue no_restart
            else_if $cur_color = purple
                set_animation idling_purple no_restart
            else_if $cur_color = white
                set_animation idling_white no_restart
            else_if $cur_color = rock
                set_animation idling_rock no_restart
            else_if $cur_color = winged
                set_animation idling_winged no_restart
            else_if $cur_color = ice
                set_animation idling_ice no_restart
            end_if
        }
        on_damage {
            play_sound squeak
        }
        on_timer {
            set_state picking_next_color
        }
    }

    dying {
        on_enter {
            set_animation dying
            set_huntable false
            start_particles enemy_defeat 0 0 150
            start_particles pellet_posy_5_petals 64 0 150
            start_dying
            if $cur_color = red
                spawn red
            else_if $cur_color = yellow
                spawn yellow
            else_if $cur_color = blue
                spawn blue
            else_if $cur_color = purple
                spawn purple
            else_if $cur_color = white
                spawn white
            else_if $cur_color = rock
                spawn rock
            else_if $cur_color = winged
                spawn winged
            else_if $cur_color = ice
                spawn ice
            end_if
        }
        on_animation_end {
            set_state reviving
        }
    }

    reviving {
        on_enter {
            set_timer $revival_time
        }
        on_timer {
            get_mob_info temp self max_health
            set_health $temp
            set_state sprout
        }
    }
}
