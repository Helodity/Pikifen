first_state = idling
death_state = destroyed

script {
    idling {
        on_enter {
            get_random_int temp 1 3			
			if $temp = 1
				set_animation idling_01
			else_if $temp = 2
				set_animation idling_02
			else_if $temp = 3
				set_animation idling_03
			end_if
            get_random_float temp 0 360
            turn_to_relative $temp
        }
        on_damage {
            play_sound hit
        }
    }
    destroyed {
        on_enter {
			start_particles spore_explosion_small 0 0 0
            set_animation destroyed
            play_sound destroy
        }
        on_animation_end {
			finish_dying
            delete
        }
    }
}
