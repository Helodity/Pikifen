first_state = idling
death_state = dying

script {
    idling {
        on_enter {
            set_var ignoring false
            set_animation idling random_time_on_spawn
            stop
            set_near_reach search
			get_random_decimal t 1 6
			set_timer $t
        }
        on_opponent_in_reach {
            if $ignoring = false
                focus trigger
                set_state chasing
            end_if
        }
        on_itch {
            set_state shaking
        }
		on_timer {
			set_state wandering
		}
		on_pikmin_land {
			get_info pikmin mob_type
			if $pikmin = Purple Pikmin
				add_health -130
			end_if
		}
    }
	
    wandering {
        on_enter {
            get_random_decimal x -500 500
            get_random_decimal y -500 500
            move_to_relative $x $y
            set_animation moving
            get_random_decimal t 1 2
            set_timer $t
        }
        on_opponent_in_reach {
            if $ignoring = false
                focus trigger
                set_state chasing
            end_if
        }
        on_reach_destination {
            set_state idling
        }
        on_timer {
			get_random_int choice 1 3
			if $choice <= 2
				set_state wandering
			else
				set_state idling
			end_if
        }
        on_far_from_home {
            set_state returning
        }
		on_pikmin_land {
			get_info pikmin mob_type
			if $pikmin = Purple Pikmin
				add_health -130
			end_if
		}
    }
    
    chasing {
        on_enter {
            set_animation moving
            move_to_target focused_mob
            set_near_reach attack
            set_far_reach chase
        }
        on_opponent_in_reach {
            set_state attacking
        }
        on_focus_off_reach {
            set_state returning
        }
        on_far_from_home {
            set_var ignoring true
            set_state returning
        }
        on_itch {
            set_state shaking
        }
		on_pikmin_land {
			get_info pikmin mob_type
			if $pikmin = Purple Pikmin
				add_health -130
			end_if
		}
    }
    
    attacking {
        on_enter {
            stop
            set_animation attacking
            start_chomping 2 mouth
        }
        on_animation_end {
            stop_chomping
            get_info t chomped_pikmin
            if $t > 0
                set_state eating
            else
                set_state idling
            end_if
        }
		on_pikmin_land {
			get_info pikmin mob_type
			if $pikmin = Purple Pikmin
				add_health -130
			end_if
		}
    }
    
    eating {
        on_enter {
            stop_chomping
            set_animation eating
        }
        on_animation_end {
            swallow_all
            set_state idling
        }
		on_pikmin_land {
			get_info pikmin mob_type
			if $pikmin = Purple Pikmin
				add_health -130
			end_if
		}
    }
    
    shaking {
        on_enter {
            stop
            set_animation shaking
        }
        on_animation_end {
            set_state idling
        }
		on_pikmin_land {
			get_info pikmin mob_type
			if $pikmin = Purple Pikmin
				add_health -130
			end_if
		}
    }
    
    returning {
        on_enter {
            move_to_target home
            set_animation moving
            set_timer 2
            set_near_reach search
        }
        on_leave {
            set_timer 0
        }
        on_timer {
            set_var ignoring false
        }
        on_reach_destination {
            set_state idling
        }
        on_opponent_in_reach {
            if $ignoring = false
                focus trigger
                set_state chasing
            end_if
        }
		on_pikmin_land {
			get_info pikmin mob_type
			if $pikmin = Purple Pikmin
				add_health -130
			end_if
		}
    }
    
    dying {
        on_enter {
            start_dying
            set_animation dying
        }
        on_animation_end {
            finish_dying
        }
    }
    
}
